<script>
  const noBtn = document.getElementById('noBtn');
  const yesBtn = document.getElementById('yesBtn');
  const kitten = document.getElementById('kitten');
  const speech = document.getElementById('speech');
  const mainScreen = document.getElementById('mainScreen');
  const winScreen = document.getElementById('winScreen');
  const winSpeech = document.getElementById('winSpeech');
  const winTitle = document.getElementById('winTitle');

  // Messages for each "No" click
  const noMessages = [
    "please 🥺",
    "itna bhi kya gussa karna, maan bhi jao nah 😢",
    "please nah, ban jao nah mera frnd 😭",
    "i promise, main bahut acha kheyal rakhunga apka, ab maan bhi jao nah 🥹",
    "jao meko ab baat nehi karni 😤"
  ];

  let noClickCount = 0;
  let cryingTimeout;

  // Move "No" button to a safe random location within the screen
  function moveNoButton() {
    const btnRect = noBtn.getBoundingClientRect();
    const btnWidth = btnRect.width;
    const btnHeight = btnRect.height;
    const padding = 20;

    // Make sure button stays within viewport
    const maxX = window.innerWidth - btnWidth - padding;
    const maxY = window.innerHeight - btnHeight - padding;

    const randomX = Math.max(padding, Math.random() * maxX);
    const randomY = Math.max(padding, Math.random() * maxY);

    noBtn.classList.add('moving');
    noBtn.style.left = randomX + 'px';
    noBtn.style.top = randomY + 'px';
  }

  // Handle "No" button click
  noBtn.addEventListener('click', handleNoClick);

  function handleNoClick(e) {
    e.preventDefault();

    // From 5th click onwards - kitten is upset and ignores
    if (noClickCount >= 4) {
      speech.textContent = noMessages[4]; // "jao meko ab baat nehi karni"
      kitten.classList.remove('crying');
      kitten.classList.add('upset');

      // Disable the yes button visual encouragement, but still allow Yes
      // Move the No button anyway
      moveNoButton();
      noClickCount++;
      return;
    }

    // Show appropriate message
    speech.textContent = noMessages[noClickCount];

    // Make kitten cry
    kitten.classList.remove('upset');
    kitten.classList.add('crying');

    // Increase Yes button size as encouragement
    const currentSize = parseFloat(window.getComputedStyle(yesBtn).fontSize);
    if (currentSize < 32) {
      yesBtn.style.fontSize = (currentSize + 3) + 'px';
      yesBtn.style.padding = (14 + noClickCount * 2) + 'px ' + (38 + noClickCount * 4) + 'px';
    }

    // Move No button
    moveNoButton();

    noClickCount++;

    // Stop crying after some time if user doesn't interact
    clearTimeout(cryingTimeout);
    cryingTimeout = setTimeout(() => {
      if (noClickCount < 5) {
        kitten.classList.remove('crying');
      }
    }, 4000);
  }

  // Handle "Yes" button click
  yesBtn.addEventListener('click', () => {
    // Special message if user agrees on 2nd or 3rd try (1 or 2 No clicks)
    if (noClickCount >= 1 && noClickCount <= 2) {
      winTitle.textContent = "Thank You! 💕";
      winSpeech.textContent = "kitni jaldi maan gaye appp, thanks! 💖✨";
    } else if (noClickCount === 0) {
      winTitle.textContent = "YAY! 🎉";
      winSpeech.textContent = "We are best friends forever! 💕🐾";
    } else if (noClickCount >= 3) {
      winTitle.textContent = "Finally! 🥰";
      winSpeech.textContent = "Aap mere sabse ache dost ho! Thank you! 💖🐾";
    }

    mainScreen.style.display = 'none';
    winScreen.classList.add('show');
    launchConfetti();
  });

  // Confetti animation
  function launchConfetti() {
    const colors = ['#ff5b8a', '#ffb3c6', '#66bb6a', '#ffd700', '#87ceeb', '#ff9a9e', '#fcb69f'];
    for (let i = 0; i < 120; i++) {
      setTimeout(() => {
        const confetti = document.createElement('div');
        confetti.className = 'confetti';
        confetti.style.left = Math.random() * 100 + 'vw';
        confetti.style.background = colors[Math.floor(Math.random() * colors.length)];
        confetti.style.animationDuration = (Math.random() * 2 + 2) + 's';
        confetti.style.borderRadius = Math.random() > 0.5 ? '50%' : '0';
        confetti.style.width = (Math.random() * 8 + 6) + 'px';
        confetti.style.height = (Math.random() * 8 + 6) + 'px';
        document.body.appendChild(confetti);
        setTimeout(() => confetti.remove(), 5000);
      }, i * 25);
    }
  }

  // Floating background decorations
  function createFloat() {
    const float = document.createElement('div');
    float.className = 'float-bg';
    float.textContent = ['💕', '💖', '🌸', '✨', '🐾', '⭐'][Math.floor(Math.random() * 6)];
    float.style.left = Math.random() * 100 + 'vw';
    float.style.fontSize = (Math.random() * 20 + 18) + 'px';
    float.style.animationDuration = (Math.random() * 5 + 6) + 's';
    document.body.appendChild(float);
    setTimeout(() => float.remove(), 12000);
  }
  setInterval(createFloat, 700);

  // Reposition no button if window resizes
  window.addEventListener('resize', () => {
    if (noBtn.classList.contains('moving')) {
      const btnRect = noBtn.getBoundingClientRect();
      if (btnRect.right > window.innerWidth || btnRect.bottom > window.innerHeight) {
        moveNoButton();
      }
    }
  });
